# Elaborar um algoritmo para calcular a média aritmética de duas notas e mostrar resultado. 
# Em seguida, faça a codificação em Python.
print('Digite dois números para saber a média aritmética de duas notas e mostrar o resultado!!!')
n1 = float (input("Informe o primeiro número:"))
n2 = float (input("Informe o segundo número:"))
media = (n1+n2)/2

print("A média dos dois números é: ",media)
