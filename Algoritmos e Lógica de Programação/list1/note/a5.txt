# Faça um algoritmo que leia a cotação do dólar no dia e a quantidade de dólares que 
# a pessoa deseja trocar por reais, calcule e exiba a quantidade correspondente em reais.
# Dólar hoje: 5,42
dolar = 5.42 
conversao = float (input("Digite a quantidade de dólares que você deseja trocar por reais: "))

cotacao = conversao * dolar

print(cotacao)
